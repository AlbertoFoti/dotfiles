cmake_minimum_required(VERSION 3.28.0)
project(guided_entry VERSION 0.1.0 LANGUAGES C CXX)

include(CTest)
enable_testing()

add_subdirectory(E2E)
add_subdirectory(Scheduler)

include_directories(E2E)
include_directories(Scheduler)
include_directories(Configuration)

add_executable(guided_entry main.cpp)

target_link_libraries(${PROJECT_NAME} e2e)
target_link_libraries(${PROJECT_NAME} scheduler)

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
